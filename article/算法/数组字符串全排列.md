## 题目

给定一个数组，其中元素都是字符串，求出这个数组中所有字符串全排列拼接结果

## 代码

```
/*
 * 返回数组中的字符串的全排列
 */
function full(arr, prefix, result) {
    if (arr.length == 1) {
        result.push(prefix + arr[0]);
        return;
    }
    arr.forEach(function (item, index) {
        var temp = arr.slice(0);
        temp.splice(index, 1);
        full(temp, prefix + item, result);
    });
}

var arr = ['1', '2', '3'];
var result = [];

full(arr, '', result);
```
